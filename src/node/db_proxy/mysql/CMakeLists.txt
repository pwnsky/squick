set(ProjectName "db_proxy_mysql")
file(GLOB soruces *.cc)
file(GLOB headers *.h)
add_library(${ProjectName} SHARED ${soruces} ${headers})

if (APPLE)
	target_link_libraries(${ProjectName} libmysqlcppconn8.dylib)
elseif(UNIX)
	target_link_libraries(${ProjectName} libmysqlcppconn8.so)
else()
	target_link_libraries(${ProjectName} mysqlcppconn8.lib)
	target_link_libraries(${ProjectName} mysqlcppconn.lib)
endif()

LinkStruct(${ProjectName})
LinkCore(${ProjectName})
SquickOut(${ProjectName} "node/db_proxy" ${SQUICK_BUILD_PLUGIN_SERVER_PATH}/db_proxy "mysql")
